    src/errorhandler.cpp
    src/file.cpp 
    src/levenshtein.cpp 
    src/lexer.cpp 
    src/parser.cpp 
    src/runargs.cpp
    src/spinner.cpp
    src/cursor.cpp

    # for better IDE support
    src/errorhandler.hpp
    src/file.hpp
    src/levenshtein.hpp
    src/lexer.hpp
    src/parser.hpp
    src/runargs.hpp
    src/tokens.hpp
    src/argparse.hpp
    src/indicators.hpp
    src/spinner.hpp
    src/cursor.hpp
)
